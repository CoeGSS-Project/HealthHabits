    selectedSizeBin = np.argmax(np.random.rand() < wpSizeCDF_IT)
    size = np.random.randint(wpSizeBins_IT[selectedSizeBin], wpSizeBins_IT[selectedSizeBin+1])
    
    wpSizes.append(size)
    wpEmployed[selectedSizeBin] += size
    genPop += size
